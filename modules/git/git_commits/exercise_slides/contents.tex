\section{Exercise 1 - Commit messages}
\begin{frame}
\begin{block}{Sum over list}
    We wish to add up all non-negative numbers in a list. We have the following piece of code:
\end{block}
\begin{block}{Code}
    \lstinputlisting[]{sum_list.txt}
\end{block}
\end{frame}

\begin{frame}
\begin{block}{Code}
    \lstinputlisting[]{sum_list.txt}
\end{block}

\begin{alertblock}{Note:}
    \lstinline{for (int i = 1; i < list.length; i++)}\\
    We will miss an element in the loop!
\end{alertblock}
\end{frame}


\begin{frame}
\begin{block}{Old}
    \lstinputlisting[]{sum_list.txt}
\end{block}

\begin{block}{New}
    We change row 2 to \\
    \lstinline{for (int i = 0; i < list.length; i++)}
\end{block}
\end{frame}

\begin{frame}
\begin{question}
What would be a good commit message for this change? Why? %fbf-question
\end{question}    
\end{frame}
